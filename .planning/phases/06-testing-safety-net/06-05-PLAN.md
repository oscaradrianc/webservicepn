```csharp
using Microsoft.Extensions.DependencyInjection;
using Microsoft.Extensions.Logging;
using Moq;
using Negocio.Business;
using Negocio.Data;
using PortalNegocioWS.Services;
using Xunit;

namespace PortalNegocioWS.Tests.Jobs;

public class CronJobTests
{
    // Helper: creates a mock IScheduleConfig<T> with valid cron expression.
    // CronExpression must be non-empty — validated in ScheduledServiceExtensions.AddCronJob
    private static Mock<IScheduleConfig<T>> BuildScheduleConfig<T>()
    {
        var mock = new Mock<IScheduleConfig<T>>();
        mock.Setup(c => c.CronExpression).Returns("0 0 * * *");
        mock.Setup(c => c.TimeZoneInfo).Returns(TimeZoneInfo.Utc);
        return mock;
    }

    [Fact]
    public async Task ActualizarEstadoSolicitudJob_DoWork_WhenFactoryThrows_DoesNotPropagate()
    {
        // Arrange — factory throws to simulate Oracle unavailability
        var mockFactory = new Mock<IDataContextFactory>();
        mockFactory.Setup(f => f.Create())
            .Throws(new InvalidOperationException("No Oracle in tests"));

        var job = new ActualizarEstadoSolicitudJob(
            BuildScheduleConfig<ActualizarEstadoSolicitudJob>().Object,
            new Mock<ILogger<ActualizarEstadoSolicitudJob>>().Object,
            mockFactory.Object);

        // Act — DoWork must NOT propagate the exception (try/catch added in Task 1)
        var exception = await Record.ExceptionAsync(() =>
            job.DoWork(CancellationToken.None));

        // Assert — null means no exception escaped DoWork
        Assert.Null(exception);
    }

    [Fact]
    public async Task EnviarNotificacionInvitacionJob_DoWork_WhenFactoryThrows_DoesNotPropagate()
    {
        // Arrange — EnviarNotificacionInvitacionJob already has try/catch in DoWork
        var mockFactory = new Mock<IDataContextFactory>();
        mockFactory.Setup(f => f.Create())
            .Throws(new InvalidOperationException("No Oracle in tests"));

        // IServiceScopeFactory needed for scope.ServiceProvider.GetRequiredService<INotificacion>()
        // The factory.Create() throw happens inside the scope, which is inside the try/catch
        var mockScopeFactory = new Mock<IServiceScopeFactory>();
        var mockScope = new Mock<IServiceScope>();
        var mockProvider = new Mock<IServiceProvider>();
        mockScope.Setup(s => s.ServiceProvider).Returns(mockProvider.Object);
        mockScopeFactory.Setup(f => f.CreateScope()).Returns(mockScope.Object);
        mockProvider.Setup(p => p.GetService(typeof(INotificacion)))
            .Returns(new Mock<INotificacion>().Object);  // provide a no-op INotificacion

        var job = new EnviarNotificacionInvitacionJob(
            BuildScheduleConfig<EnviarNotificacionInvitacionJob>().Object,
            new Mock<ILogger<EnviarNotificacionInvitacionJob>>().Object,
            mockScopeFactory.Object,
            mockFactory.Object);

        // Act
        var exception = await Record.ExceptionAsync(() =>
            job.DoWork(CancellationToken.None));

        // Assert — existing try/catch in EnviarNotificacionInvitacionJob absorbs the Oracle failure
        Assert.Null(exception);
    }

    [Fact]
    public async Task NotificacionActualizacionDatosJob_DoWork_WhenNotificacionThrows_DoesNotPropagate()
    {
        // Arrange — this job has NO IDataContextFactory; uses IServiceScopeFactory only
        // Force INotificacion resolution to throw — job's try/catch should absorb it
        var mockScope = new Mock<IServiceScope>();
        var mockProvider = new Mock<IServiceProvider>();
        mockScope.Setup(s => s.ServiceProvider).Returns(mockProvider.Object);

        var mockScopeFactory = new Mock<IServiceScopeFactory>();
        mockScopeFactory.Setup(f => f.CreateScope()).Returns(mockScope.Object);

        // GetRequiredService<INotificacion>() calls GetService(typeof(INotificacion)) internally
        mockProvider.Setup(p => p.GetService(typeof(INotificacion)))
            .Throws(new InvalidOperationException("No INotificacion in tests"));

        var job = new NotificacionActualizacionDatosJob(
            BuildScheduleConfig<NotificacionActualizacionDatosJob>().Object,
            new Mock<ILogger<NotificacionActualizacionDatosJob>>().Object,
            mockScopeFactory.Object);

        // Act
        var exception = await Record.ExceptionAsync(() =>
            job.DoWork(CancellationToken.None));

        // Assert — existing try/catch in NotificacionActualizacionDatosJob absorbs the failure
        Assert.Null(exception);
    }
}
```

**Note on INotificacion mock for EnviarNotificacionInvitacionJob:** The mock scope provides a no-op `INotificacion` so the LINQ query on Oracle (which throws via `mockFactory.Create()`) is reached and caught. The `IServiceScopeFactory` mock is needed because `EnviarNotificacionInvitacionJob.DoWork` calls `_serviceScopeFactory.CreateScope()` before `_factory.Create()`.

**Note on NotificacionActualizacionDatosJob:** `GetRequiredService<T>()` calls `GetService(typeof(T))` and throws `InvalidOperationException` when it returns null. By returning a thrown exception from `GetService(typeof(INotificacion))` directly, we bypass the null-return path and force the exception that the job's catch block must absorb.
