```csharp
using System.Net;
using System.Net.Http.Headers;
using System.Net.Http.Json;
using Microsoft.AspNetCore.Mvc.Testing;
using Microsoft.Extensions.DependencyInjection;
using Moq;
using Negocio.Business;       // ISolicitudCompra
using Negocio.Model;          // Autorizacion, SolicitudCompra model — adjust if different namespace
using PortalNegocioWS.Tests.Infrastructure;
using Xunit;

namespace PortalNegocioWS.Tests.Business;

public class SolicitudTests : IClassFixture<CustomWebApplicationFactory>
{
    private readonly CustomWebApplicationFactory _factory;

    public SolicitudTests(CustomWebApplicationFactory factory)
    {
        _factory = factory;
    }

    // Helper: creates an authenticated HttpClient with ISolicitudCompra replaced by the given mock.
    // Per-test factory variant required because each test needs its own Mock<ISolicitudCompra>.
    private HttpClient CreateAuthenticatedClient(Mock<ISolicitudCompra> mockSolicitud)
    {
        var client = _factory.WithWebHostBuilder(builder =>
        {
            builder.ConfigureTestServices(services =>
            {
                // Remove real Scoped registration (BusinessInstaller line 21)
                var descriptor = services.SingleOrDefault(
                    d => d.ServiceType == typeof(ISolicitudCompra));
                if (descriptor != null) services.Remove(descriptor);
                services.AddScoped(_ => mockSolicitud.Object);
            });
        }).CreateClient();

        client.DefaultRequestHeaders.Authorization =
            new AuthenticationHeaderValue("Bearer", JwtTokenHelper.GenerateToken());
        return client;
    }

    [Fact]
    public async Task AutorizarSolicitud_WhenBusinessReturnsOK_Returns200()
    {
        // Arrange
        var mockSolicitud = new Mock<ISolicitudCompra>();
        mockSolicitud
            .Setup(s => s.AutorizarSolicitud(It.IsAny<Autorizacion>()))
            .Returns("OK");   // sync — matches string AutorizarSolicitud(Autorizacion)

        // Act
        var response = await CreateAuthenticatedClient(mockSolicitud)
            .PostAsJsonAsync("/api/Solicitud/Autorizar", new Autorizacion());

        // Assert — controller returns Ok() when result == "OK"
        Assert.Equal(HttpStatusCode.OK, response.StatusCode);
    }

    [Fact]
    public async Task RegistrarSolicitud_WhenBusinessReturnsOK_Returns200()
    {
        // Arrange
        var mockSolicitud = new Mock<ISolicitudCompra>();
        mockSolicitud
            .Setup(s => s.RegistrarSolicitud(It.IsAny<SolicitudCompra>()))
            .ReturnsAsync("OK");  // async — matches Task<string> RegistrarSolicitud(SolicitudCompra)

        // Act
        var response = await CreateAuthenticatedClient(mockSolicitud)
            .PostAsJsonAsync("/api/Solicitud/registrar", new SolicitudCompra());

        // Assert — controller returns Ok() when result == "OK"
        Assert.Equal(HttpStatusCode.OK, response.StatusCode);
    }
}
```
