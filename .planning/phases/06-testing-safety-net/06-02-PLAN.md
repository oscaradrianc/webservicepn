**Create `PortalNegocioWS.Tests/Auth/AuthBoundaryTests.cs`:**

```csharp
using System.Net;
using System.Net.Http.Headers;
using PortalNegocioWS.Tests.Infrastructure;
using Xunit;

namespace PortalNegocioWS.Tests.Auth;

public class AuthBoundaryTests : IClassFixture<CustomWebApplicationFactory>
{
    private readonly HttpClient _client;

    public AuthBoundaryTests(CustomWebApplicationFactory factory)
    {
        _client = factory.CreateClient(new WebApplicationFactoryClientOptions
        {
            AllowAutoRedirect = false   // Prevent 401→redirect masking
        });
    }

    [Fact]
    public async Task ProtectedEndpoint_WithoutToken_Returns401()
    {
        // GET /api/Solicitud/list — [Authorize] endpoint, no JWT header
        var response = await _client.GetAsync("/api/Solicitud/list?tipo=1");
        Assert.Equal(HttpStatusCode.Unauthorized, response.StatusCode);
    }

    [Fact]
    public async Task ProtectedEndpoint_WithValidToken_DoesNotReturn401()
    {
        var token = JwtTokenHelper.GenerateToken();
        _client.DefaultRequestHeaders.Authorization =
            new AuthenticationHeaderValue("Bearer", token);

        // Any non-401 is acceptable — the business layer is mocked and may return 400/500
        // The assertion only verifies that authentication middleware passed the request through
        var response = await _client.GetAsync("/api/Solicitud/list?tipo=1");
        Assert.NotEqual(HttpStatusCode.Unauthorized, response.StatusCode);
    }
}
```
