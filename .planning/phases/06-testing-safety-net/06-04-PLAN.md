```csharp
using AutoMapper;
using PortalNegocioWS.Installers;

namespace PortalNegocioWS.Tests.Mapping;

public class AutoMapperConfigTests
{
    /// <summary>
    /// Mirrors AutoMapperInstaller.InstallServices exactly (lines 10-15).
    /// Uses the same assembly scan so any Profile added later is automatically included.
    /// Throws AutoMapperConfigurationException if:
    ///   - A destination property has no mapping and no Ignore()
    ///   - A custom value resolver type is wrong
    ///   - A circular reference is misconfigured
    /// </summary>
    [Fact]
    public void AutoMapper_AllProfiles_ConfigurationIsValid()
    {
        // Mirror of AutoMapperInstaller lines 10-13 — same scan, same assembly
        var config = new MapperConfiguration(c =>
        {
            c.AddMaps(typeof(AutoMapperInstaller).Assembly);
        });

        // Throws AutoMapperConfigurationException if any Profile has a broken mapping.
        // This is the same call already in production startup — this test makes it a CI gate.
        config.AssertConfigurationIsValid();
    }
}
```

No `WebApplicationFactory`, no `HttpClient`, no mocks. This is a pure unit test.
