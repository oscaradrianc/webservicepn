**PortalNegocioWS/Handlers/GlobalExceptionHandler.cs**

Requirements:
1. Implement `IExceptionHandler` from `Microsoft.AspNetCore.Diagnostics`
2. Constructor-inject `ILogger<GlobalExceptionHandler>` and `IProblemDetailsService`
3. In `TryHandleAsync`: log the exception first with `_logger.LogError(exception, "Unhandled exception: {Message}", exception.Message)`, then determine the HTTP status code via switch expression on exception type:
   - `BusinessException` → 400
   - `NotFoundException` → 404
   - `UnauthorizedException` → 401
   - any other exception → 500 (use generic "An unexpected error occurred." as the detail — NEVER expose exception.Message for non-business exceptions in production)
4. Set `httpContext.Response.StatusCode` to the determined status code
5. Write the response using `IProblemDetailsService.WriteAsync()` passing a `ProblemDetailsContext` — this routes through ASP.NET Core's content negotiation pipeline and respects the Newtonsoft.Json formatter configuration
6. Return `true` from `TryHandleAsync`
7. Namespace: `PortalNegocioWS.Handlers`

```csharp
using Microsoft.AspNetCore.Diagnostics;
using Microsoft.AspNetCore.Mvc;
using Microsoft.AspNetCore.Mvc.Infrastructure;
using PortalNegocioWS.Exceptions;

namespace PortalNegocioWS.Handlers
{
    public class GlobalExceptionHandler : IExceptionHandler
    {
        private readonly ILogger<GlobalExceptionHandler> _logger;
        private readonly IProblemDetailsService _problemDetailsService;

        public GlobalExceptionHandler(
            ILogger<GlobalExceptionHandler> logger,
            IProblemDetailsService problemDetailsService)
        {
            _logger = logger;
            _problemDetailsService = problemDetailsService;
        }

        public async ValueTask<bool> TryHandleAsync(
            HttpContext httpContext,
            Exception exception,
            CancellationToken cancellationToken)
        {
            _logger.LogError(exception, "Unhandled exception: {Message}", exception.Message);

            var (statusCode, title, detail) = exception switch
            {
                BusinessException ex => (StatusCodes.Status400BadRequest, "Bad Request", ex.Message),
                NotFoundException ex => (StatusCodes.Status404NotFound, "Not Found", ex.Message),
                UnauthorizedException ex => (StatusCodes.Status401Unauthorized, "Unauthorized", ex.Message),
                _ => (StatusCodes.Status500InternalServerError, "Internal Server Error", "An unexpected error occurred.")
            };

            httpContext.Response.StatusCode = statusCode;

            await _problemDetailsService.WriteAsync(new ProblemDetailsContext
            {
                HttpContext = httpContext,
                ProblemDetails =
                {
                    Status = statusCode,
                    Title = title,
                    Detail = detail
                }
            });

            return true;
        }
    }
}
```
